export function urgencyText(
  serviceName: string,
  days: number,
  amount: number,
  theme: 'genz' | 'millennial' | 'genx'
): { title: string; subtitle: string } {
  if (theme === 'genz') {
    return {
      title: `${serviceName} drops in ${days} days 🔥`,
      subtitle: `$${amount} incoming — cancel or keep?`
    }
  }
  if (theme === 'millennial') {
    return {
      title: `${serviceName} renews in ${days} days`,
      subtitle: `$${amount} will be charged`
    }
  }
  return {
    title: `RENEWAL: ${serviceName.toUpperCase()}`,
    subtitle: `$${amount} | ${days}D | ACTION REQUIRED`
  }
}
